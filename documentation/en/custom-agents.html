Python Example

import requests

class IAFactoryAgent:
    def __init__(self, api_key, agent_config):
        self.api_key = api_key
        self.config = agent_config
        self.url = "https://api.iafactory-algeria.com/v1/chat/completions"
    
    def run(self, user_input):
        """Execute the agent with user input"""
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        
        data = {
            "model": self.config['model'],
            "temperature": self.config['temperature'],
            "max_tokens": self.config['max_tokens'],
            "messages": [
                {
                    "role": "system",
                    "content": self.config['system_prompt']
                },
                {
                    "role": "user",
                    "content": user_input
                }
            ]
        }
        
        response = requests.post(self.url, headers=headers, json=data)
        return response.json()['choices'][0]['message']['content']

Multi-Agent Workflow

class AgentWorkflow:
    def __init__(self, api_key):
        self.api_key = api_key
        self.agents = {}
    
    def add_agent(self, name, config):
        """Add an agent to the workflow"""
        self.agents[name] = IAFactoryAgent(self.api_key, config)
    
    def run_pipeline(self, input_data):
        """Execute a chain of agents"""
        result = input_data
        
        # Agent 1: Extraction
        result = self.agents['extractor'].run(result)
        
        # Agent 2: Analysis
        result = self.agents['analyst'].run(result)
        
        # Agent 3: Writing
        final_output = self.agents['writer'].run(result)
        
        return final_output

# Example: Customer feedback processing pipeline
workflow = AgentWorkflow("your_api_key")

workflow.add_agent('extractor', {
    "model": "gpt-4.1-mini",
    "temperature": 0.1,
    "max_tokens": 2000,
    "system_prompt": "Extract key points and sentiments from customer feedback"
})

workflow.add_agent('analyst', {
    "model": "gpt-4.1",
    "temperature": 0.3,
    "max_tokens": 2000,
    "system_prompt": "Analyze patterns and trends in feedback"
})

workflow.add_agent('writer', {
    "model": "gpt-4.1",
    "temperature": 0.5,
    "max_tokens": 3000,
    "system_prompt": "Write an executive report with recommendations"
})

# Usage
feedbacks = """
[100+ customer feedbacks]
"""

report = workflow.run_pipeline(feedbacks)
print(report)

‚ú® Best Practices

1. Effective Prompts

‚úÖ Good: 
You are an expert in [specific domain].
Objective: [precise task]
Constraints: [detailed list]
Expected format: [clear structure]
‚ùå Bad: 
You are an assistant. Help me.

2. Context Management

  • Limit context size (< 8000 tokens)
  • Prioritize relevant information
  • Use summaries for long documents
  • Implement caching to save costs

3. Monitoring and Improvement

import logging

class MonitoredAgent(IAFactoryAgent):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.metrics = {
            'calls': 0,
            'errors': 0,
            'total_tokens': 0
        }
    
    def run(self, user_input):
        self.metrics['calls'] += 1
        
        try:
            result = super().run(user_input)
            # Log metrics
            logging.info(f"Agent {self.config['name']}: Success")
            return result
        except Exception as e:
            self.metrics['errors'] += 1
            logging.error(f"Agent error: {e}")
            raise
    
    def get_metrics(self):
        return self.metrics
