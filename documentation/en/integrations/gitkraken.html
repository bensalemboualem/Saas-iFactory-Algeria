IAFactory API Setup

Create a Git hook script with IAFactory:

# .git/hooks/prepare-commit-msg
#!/bin/bash

# Get staged changes
DIFF=$(git diff --cached)

# Call IAFactory to generate commit message
COMMIT_MSG=$(curl -s -X POST https://api.iafactory-algeria.com/v1/chat/completions \
  -H "Authorization: Bearer $IAFACTORY_API_KEY" \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"gpt-4.1-mini\",
    \"messages\": [{
      \"role\": \"system\",
      \"content\": \"Generate a concise git commit message in conventional commits format (type(scope): description). Be specific about what changed.\"
    }, {
      \"role\": \"user\",
      \"content\": \"Git diff:\n$DIFF\"
    }],
    \"temperature\": 0.3,
    \"max_tokens\": 100
  }" | jq -r '.choices[0].message.content')

# Write to commit file
echo "$COMMIT_MSG" > "$1"
# Make executable
chmod +x .git/hooks/prepare-commit-msg

2. Pull Request Review

Script for automatic PR review:

#!/bin/bash
# review-pr.sh

PR_NUMBER=$1
REPO_PATH=$2

# Get PR diff
cd "$REPO_PATH"
git fetch origin pull/$PR_NUMBER/head:pr-$PR_NUMBER
git diff main...pr-$PR_NUMBER > pr-diff.txt

# Analyze with IAFactory
curl -X POST https://api.iafactory-algeria.com/v1/chat/completions \
  -H "Authorization: Bearer $IAFACTORY_API_KEY" \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"gpt-4.1\",
    \"messages\": [{
      \"role\": \"system\",
      \"content\": \"You are an expert code reviewer. Analyze this PR and provide:\n1. Overall assessment\n2. Security issues (if any)\n3. Performance concerns\n4. Best practices violations\n5. Suggested improvements\n\nFormat in Markdown.\"
    }, {
      \"role\": \"user\",
      \"content\": \"$(cat pr-diff.txt)\"
    }],
    \"temperature\": 0.2,
    \"max_tokens\": 2000
  }" | jq -r '.choices[0].message.content' > pr-review.md

# Display review
cat pr-review.md

3. Conflict Resolution

#!/bin/bash
# resolve-conflict.sh

CONFLICT_FILE=$1

# Extract conflict sections
CONFLICT_CONTENT=$(cat "$CONFLICT_FILE")

# Ask AI to resolve
curl -X POST https://api.iafactory-algeria.com/v1/chat/completions \
  -H "Authorization: Bearer $IAFACTORY_API_KEY" \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"gpt-4.1\",
    \"messages\": [{
      \"role\": \"system\",
      \"content\": \"Resolve this git merge conflict. Keep the best parts from both versions. Return ONLY the resolved code, no explanations.\"
    }, {
      \"role\": \"user\",
      \"content\": \"$CONFLICT_CONTENT\"
    }],
    \"temperature\": 0.1
  }" | jq -r '.choices[0].message.content' > "$CONFLICT_FILE.resolved"

echo "Suggested resolution in $CONFLICT_FILE.resolved"
echo "Review before applying!"

4. Automatic Changelog

#!/bin/bash
# generate-changelog.sh

FROM_TAG=$1
TO_TAG=$2

# Get commits between tags
git log $FROM_TAG..$TO_TAG --oneline > commits.txt

# Generate changelog with AI
curl -X POST https://api.iafactory-algeria.com/v1/chat/completions \
  -H "Authorization: Bearer $IAFACTORY_API_KEY" \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"gpt-4.1\",
    \"messages\": [{
      \"role\": \"system\",
      \"content\": \"Transform these git commits into a clean CHANGELOG.md format. Group by: Features, Bug Fixes, Performance, Documentation, Refactoring.\"
    }, {
      \"role\": \"user\",
      \"content\": \"$(cat commits.txt)\"
    }],
    \"temperature\": 0.3,
    \"max_tokens\": 1500
  }" | jq -r '.choices[0].message.content' > CHANGELOG-$TO_TAG.md

echo "Changelog generated: CHANGELOG-$TO_TAG.md"

Release Workflow

#!/bin/bash
# release.sh

VERSION=$1

# 1. Generate changelog
./scripts/generate-changelog.sh $(git describe --tags --abbrev=0) HEAD

# 2. Generate release notes
curl -X POST https://api.iafactory-algeria.com/v1/chat/completions \
  -H "Authorization: Bearer $IAFACTORY_API_KEY" \
  -H "Content-Type: application/json" \
  -d "{
    \"model\": \"gpt-4.1\",
    \"messages\": [{
      \"role\": \"system\",
      \"content\": \"Create release notes from this changelog. Highlight major features, breaking changes, and migration guides.\"
    }, {
      \"role\": \"user\",
      \"content\": \"$(cat CHANGELOG-HEAD.md)\"
    }]
  }" | jq -r '.choices[0].message.content' > RELEASE-NOTES-$VERSION.md

# 3. Tag and push
git tag -a v$VERSION -m "Release v$VERSION"
git push origin v$VERSION

# 4. Create GitHub release
gh release create v$VERSION --notes-file RELEASE-NOTES-$VERSION.md

echo "âœ… Release v$VERSION created!"
