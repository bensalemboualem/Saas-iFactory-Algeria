üì¶ Setup

1. Install Xcode Extension

# Clone the extension repository
git clone https://github.com/iafactory-algeria/xcode-iafactory.git
cd xcode-iafactory

# Build and install
xcodebuild -scheme IAFactoryExtension
open build/IAFactoryExtension.app

2. Enable Extension

  1. Open System Settings ‚Üí Privacy & Security ‚Üí Extensions
  2. Enable Xcode Source Editor
  3. Check IAFactory Extension
  4. Restart Xcode

3. Configure API Key

# Set API key in environment
echo 'export IAFACTORY_API_KEY="your_api_key"' >> ~/.zshrc
source ~/.zshrc

# Or configure in Xcode
defaults write com.iafactory.xcode-extension apiKey "your_api_key"

// Generate a SwiftUI view with a list of users fetched from API

// AI generates:
import SwiftUI

struct UserListView: View {
    @State private var users: [User] = []
    @State private var isLoading = false
    @State private var errorMessage: String?
    
    var body: some View {
        NavigationView {
            Group {
                if isLoading {
                    ProgressView("Loading users...")
                } else if let error = errorMessage {
                    VStack {
                        Image(systemName: "exclamationmark.triangle")
                            .font(.largeTitle)
                        Text(error)
                            .foregroundColor(.red)
                        Button("Retry") {
                            fetchUsers()
                        }
                    }
                } else {
                    List(users) { user in
                        NavigationLink(destination: UserDetailView(user: user)) {
                            UserRowView(user: user)
                        }
                    }
                }
            }
            .navigationTitle("Users")
            .onAppear {
                fetchUsers()
            }
        }
    }
    
    private func fetchUsers() {
        isLoading = true
        errorMessage = nil
        
        guard let url = URL(string: "https://api.example.com/users") else {
            errorMessage = "Invalid URL"
            isLoading = false
            return
        }
        
        URLSession.shared.dataTask(with: url) { data, response, error in
            DispatchQueue.main.async {
                isLoading = false
                
                if let error = error {
                    errorMessage = error.localizedDescription
                    return
                }
                
                guard let data = data else {
                    errorMessage = "No data received"
                    return
                }
                
                do {
                    users = try JSONDecoder().decode([User].self, from: data)
                } catch {
                    errorMessage = "Failed to decode: \(error.localizedDescription)"
                }
            }
        }.resume()
    }
}

Unit Test Generation

Select a function ‚Üí IAFactory ‚Üí Generate Tests

// Original function
func calculateDiscount(price: Double, percentage: Double) throws -> Double {
    guard percentage >= 0 && percentage <= 100 else {
        throw DiscountError.invalidPercentage
    }
    return price * (1 - percentage / 100)
}

// Generated tests
import XCTest
@testable import YourApp

class DiscountCalculatorTests: XCTestCase {
    
    func testCalculateDiscountWithValidPercentage() throws {
        let result = try calculateDiscount(price: 100, percentage: 10)
        XCTAssertEqual(result, 90, accuracy: 0.01)
    }
    
    func testCalculateDiscountWithZeroPercentage() throws {
        let result = try calculateDiscount(price: 100, percentage: 0)
        XCTAssertEqual(result, 100, accuracy: 0.01)
    }
    
    func testCalculateDiscountWithFullDiscount() throws {
        let result = try calculateDiscount(price: 100, percentage: 100)
        XCTAssertEqual(result, 0, accuracy: 0.01)
    }
    
    func testCalculateDiscountWithNegativePercentage() {
        XCTAssertThrowsError(try calculateDiscount(price: 100, percentage: -10)) { error in
            XCTAssertEqual(error as? DiscountError, .invalidPercentage)
        }
    }
    
    func testCalculateDiscountWithPercentageOver100() {
        XCTAssertThrowsError(try calculateDiscount(price: 100, percentage: 150)) { error in
            XCTAssertEqual(error as? DiscountError, .invalidPercentage)
        }
    }
    
    func testCalculateDiscountWithLargeNumbers() throws {
        let result = try calculateDiscount(price: 1_000_000, percentage: 25)
        XCTAssertEqual(result, 750_000, accuracy: 0.01)
    }
}

Error Explanation

Click on error ‚Üí IAFactory ‚Üí Explain Error

// Error: Type 'ViewController' does not conform to protocol 'UITableViewDataSource'

// AI Explanation:
/*
This error occurs because ViewController declares conformance to UITableViewDataSource
but doesn't implement all required methods.

Required methods to implement:
1. tableView(_:numberOfRowsInSection:) -> Int
2. tableView(_:cellForRowAt:) -> UITableViewCell

Quick fix:
*/

extension ViewController: UITableViewDataSource {
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return items.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "Cell", for: indexPath)
        cell.textLabel?.text = items[indexPath.row]
        return cell
    }
}
